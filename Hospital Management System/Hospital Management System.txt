int login(){
    char user[20], pass[20];
    char saved_user[20], saved_pass[20];

    printf("Set Username: ");
    scanf("%s", saved_user);

    printf("Set Password: ");
    scanf("%s", saved_pass);

    printf("\n--- Login ---\n");

    printf("Enter Username: ");
    scanf("%s", user);

    printf("Enter Password: ");
    scanf("%s", pass);

    if(strcmp(user, saved_user)==0 && strcmp(pass, saved_pass)==0){
        printf("Login Successful!\n");
        return 1;
    } else {
        printf("Wrong login!\n");
        return 0;
    }
}


struct Doctor{
	int doc_id;
	char name[50];
	char specialization[50];
}d;


void addDoc(){
	FILE *fp;
	fp=fopen("doctor.txt","a");
	printf("Enter Doctor ID: ");
	scanf("%d",&d.doc_id);
	printf("Enter Doctor's name: ");
	scanf("%s",d.name);
	printf("Enter Specialization: ");
	scanf("%s",d.specialization);
	
	fwrite(&d,sizeof(d),1,fp);
	fclose(fp);
	printf("Doctor added successfully!\n");
}
void viewDoc(){
	FILE *fp=fopen("doctor.txt","r");
	printf("\n--- List of Doctors ---\n");
	while(fread(&d, sizeof(d),1,fp))
	{
		printf("\nID: %d",d.doc_id);
		printf("\nName: %s", d.name);
		printf("\nSpecialization: %s\n",d.specialization);
	}
	fclose(fp);
}
void searchDoc(){
	FILE *fp=fopen("doctor.txt","r");
	int id, found=0;
	printf("Enter Doctor ID: ");
	scanf("%d",&id);
	while(fread(&d,sizeof(d),1,fp)){
		if(d.doc_id==id){
			printf("%d %s %s\n",d.doc_id,d.name,d.specialization);
			found=1;
		}
	}
	if(!found)
		printf("Not found\n");
	fclose(fp);
}
void delDoc(){
	FILE *fp=fopen("doctor.txt","r");
	FILE *temp=fopen("temp.txt","w");
	int id;
	printf("Enter Doctor ID: ");
	scanf("%d",&id);
	while(fread(&d,sizeof(d),1,fp)){
		if(d.doc_id!=id)
			fwrite(&d,sizeof(d),1,temp);
	}
	fclose(fp);fclose(temp);
	remove("doctor.txt");
	rename("temp.txt","doctor.txt");
	printf("Deleted\n");
}
void updateDoc(){
    FILE *fp = fopen("doctor.txt","r");
    FILE *temp = fopen("temp.txt","w");

    int id, found = 0;

    printf("Enter Doctor ID: ");
    scanf("%d",&id);

    while(fread(&d,sizeof(d),1,fp)){
        if(d.doc_id == id && found == 0){
            printf("Enter new name: ");
            scanf("%s", d.name);

            printf("Enter specialization: ");
            scanf("%s", d.specialization);

            found = 1;   // mark as updated
        }

        fwrite(&d,sizeof(d),1,temp);
    }

    fclose(fp);
    fclose(temp);

    remove("doctor.txt");
    rename("temp.txt","doctor.txt");

    if(found)
        printf("Record Updated Successfully!\n");
    else
        printf("Doctor not found!\n");
}

struct Visit {
    int p_id;
    int doc_id;
} v;
void assignDoc() {
    FILE *fp = fopen("visit.txt", "a");

    printf("Enter Patient ID: ");
    scanf("%d", &v.p_id);

    printf("Enter Doctor ID: ");
    scanf("%d", &v.doc_id);

    fwrite(&v, sizeof(v), 1, fp);
    fclose(fp);

    printf("Assigned successfully\n");
}
void viewVisit() {
    FILE *fpV = fopen("visit.txt", "r");
    FILE *fpD = fopen("doctor.txt", "r");
    FILE *fpP = fopen("patient.txt", "r");

    printf("	\nDoctor - Patient List:\n");

    while (fread(&v, sizeof(v), 1, fpV)) {

        rewind(fpD);
        while (fread(&d, sizeof(d), 1, fpD)) {
            if (d.doc_id == v.doc_id) {

                rewind(fpP);
                while (fread(&p, sizeof(p), 1, fpP)) {
                    if (p.p_id == v.p_id) {

                        printf("Doctor: %s -> Patient: %s\n",
                               d.name, p.name);
                    }
                }
            }
        }
    }

    fclose(fpV);
    fclose(fpD);
    fclose(fpP);
}
